% Program to display Original, Grayscale, and Binary Images

clc;
clear all;
close all;

% Read the input image
img = imread('peppers.png');

% Store original image
originalImg = img;

%% Convert RGB to Grayscale
if size(img,3) == 3
    grayImg = rgb2gray(img);
else
    grayImg = img;
end

%% Convert Grayscale to Binary
binaryImg = imbinarize(grayImg);

%% Display all images in a single figure
figure;

% Original Image
subplot(1,3,1);
imshow(originalImg);
title('Original Image');

% Grayscale Image
subplot(1,3,2);
imshow(grayImg);
title('Gray Image');

% Binary Image
subplot(1,3,3);
imshow(binaryImg);
title('Binary Image');

% Overall title
sgtitle('Image Conversion');
